1. total sales = COUNTROWS('company order_items')
2.Total revenue = SUM('company order_items'[price])
3.Total orders = DISTINCTCOUNT('company orders'[order_id])
4.avg freight per seller = DIVIDE(SUM('company order_items'[freight_value]),DISTINCTCOUNT('company order_items'[seller_id]))
5.avg orders per seller = DIVIDE(DISTINCTCOUNT('company order_items'[freight_value]),DISTINCTCOUNT('company order_items'[seller_id]))
6.avg payment value = AVERAGE('company payments'[payment_value])
7.avg order value = DIVIDE(SUM('company order_items'[price]),DISTINCTCOUNT('company order_items'[order_id]))
8.total freight cost = SUM('company order_items'[freight_value])
9.total customers = DISTINCTCOUNT('company customers'[customer_id])
10.returning customers = CALCULATE(DISTINCTCOUNT('company orders'[customer_id]),FILTER(VALUES('company orders'[customer_id]),CALCULATE(COUNT('company orders'[order_id]))>1))
11.New Customers = CALCULATE(DISTINCTCOUNT('company orders'[customer_id]),FILTER(VALUES('company orders'[customer_id]),CALCULATE(COUNT('company orders'[order_id]))=1))
